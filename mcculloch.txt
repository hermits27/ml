

import numpy as np
from sklearn.datasets import load_iris
from sklearn.preprocessing import StandardScaler
# 1. Load dataset
data = load_iris()
X = data.data[:, :2]  # 2 features
y = data.target

# 2. Convert to binary classification
y = np.where(y == 0, 1, -1)

# 3. Normalize data
scaler = StandardScaler()
X = scaler.fit_transform(X)

# 4. Manually define weights and threshold
w = np.array([0.6, -0.4])
theta = 0.1

# 5. McCulloch-Pitts function
def mc_culloch_pitts(X, w, theta):
    outputs = []
    for x in X:
        activation = np.dot(x, w)
        if activation >= theta:
            outputs.append(1)
        else:
            outputs.append(-1)
    return np.array(outputs)

# 6. Predictions
y_pred = mc_culloch_pitts(X, w, theta)

# 7. Accuracy
accuracy = np.mean(y_pred == y)
print("Accuracy:", accuracy)
